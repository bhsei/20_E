121def cancel_job(job): 

122 """在节点上取消作业""" 

123 node = job.node 

124 if node is None: 

125 return 

126 url = f'http://{node.ip}:{node.port}/cancel.json' 

127 data = {'project': job.task.template.site.name, 'job': job.uuid}

129 requests.post(url, data, auth=(node.username, node.password), timeout=3)

142@require_http_methods(['POST']) 

143def change_task_status(request, task_pk, status): 

144 """修改任务状态""" 

145 if not request.user.is_authenticated: 

146 return redirect(reverse('login')) 

147 task = get_object_or_404(Task, pk=task_pk) 

148 if task.user_id != request.user.id: 

149 return HttpResponseForbidden('Not your task') 

150 if task.status not in ALLOWED_OPERATION or status not in ALLOWED_OPERATION[task.status]: 

151 return HttpResponseForbidden('Operation not allowed') 

152 

153 task.status = status 

154 task.save() 

155 if status == 'paused': 

156 for job in task.job_set.filter(status=JobStatus.PENDING): 

157 cancel_job(job) 

158 job.status = JobStatus.CREATED 

159 job.save() 

160 elif status == 'canceled': 

161 for job in task.job_set.exclude(status=JobStatus.FINISHED): 

162 cancel_job(job) 

163 job.status = JobStatus.FINISHED 

164 job.save() 

165 return JsonResponse({'status': 'SUCCESS'})

169def clear_data(request, task_pk): 

170 """清除任务的爬取结果数据"""

179 try: 

180 client = MongoClient(settings.MONGODB_URI) 

181 db = client['{}_{}'.format(task.template.site.name, task.template.name)] 

182 for c in db.list_collection_names(filter={'name': {'$regex': '{:d}_.*'.format(task.id)}}): 

183 db.drop_collection(c) 

184 client.close()
