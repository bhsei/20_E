102 """ 

103 向所有在线节点推送模板集合的Egg 

104 

105 :param project_name: template name to be deployed 

106 :param egg_path: absolute path to egg file on this machine 

107 """ 

108 nodes = Node.objects.filter(status=NodeStatus.ONLINE) 

109 for node in nodes: 

110 try: 

111 url = f'http://{node.ip}:{node.port}/delproject.json' 

112 data = {'project': project_name} 

113 requests.post(url, data=data, auth=(node.username, node.password), timeout=3) 

114 url = f'http://{node.ip}:{node.port}/addversion.json' 

115 data = {'project': project_name, 'version': 'r1'} 

116 requests.post(url, files={'egg': open(egg_path, 'rb')}, data=data, auth=( 

117 node.username, node.password)) 

118 except requests.exceptions.RequestException as _e: 

119 node.status = NodeStatus.OFFLINE 

120 node.save()
